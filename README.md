### Mi configuración VIM

set nocompatible

set ruler

set nu

set wrap

set tabstop=2

set sw=2

set cindent

set autoindent

set smartindent

set copyindent

set bs=2

set columns=80

set background=dark

set wrapmargin=4

set nobk

syntax on

set sm

set noexpandtab

colorscheme monokai-phoenix

set enc=utf-8

set foldmethod=indent

set fileencoding=utf-8

set fileencodings=ucs-bom,utf8,prc

set cursorline

set cursorcolumn

set gfn=Monospace:h12:b

set mouse=a

set incsearch

set hlsearch

     **Comando principal** :MinimapSync
